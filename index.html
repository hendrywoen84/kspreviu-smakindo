function doPost(e) {
  var lock = LockService.getScriptLock();
  lock.tryLock(10000);
  
  try {
    var data = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheetName = data.segment || 'A';
    var sheet = ss.getSheetByName(sheetName);
    
    // CEK APakah SUDAH PERNAH SUBMIT
    var alreadySubmitted = checkIfAlreadySubmitted(ss, data.name, sheetName);
    if (alreadySubmitted) {
      return ContentService.createTextOutput(
        JSON.stringify({
          status: 'duplicate',
          message: 'Anda sudah mengisi survey ini sebelumnya.'
        })
      ).setMimeType(ContentService.MimeType.JSON);
    }
    
    if (!sheet) {
      sheet = ss.insertSheet(sheetName);
      var headers = ['Timestamp', 'Nama Responden', 'Segmen', 'Status'];
      Object.keys(data.answers || {}).forEach(function(key) {
        headers.push(key);
      });
      sheet.appendRow(headers);
    }
    
    var timestamp = new Date();
    var row = [timestamp, data.name || 'Unknown', sheetName, 'COMPLETED'];
    
    Object.keys(data.answers || {}).forEach(function(key) {
      var jawaban = data.answers[key];
      if (Array.isArray(jawaban)) {
        row.push(jawaban.join(', '));
      } else {
        row.push(jawaban || '');
      }
    });
    
    sheet.appendRow(row);
    
    var logSheet = ss.getSheetByName('Log');
    if (!logSheet) {
      logSheet = ss.insertSheet('Log');
      logSheet.appendRow(['Timestamp', 'Event', 'Nama', 'Segmen', 'Status']);
    }
    logSheet.appendRow([timestamp, 'SUBMIT_SUCCESS', data.name, sheetName, 'First Submission']);
    
    return ContentService.createTextOutput(
      JSON.stringify({status: 'success', message: 'Data berhasil disimpan'})
    ).setMimeType(ContentService.MimeType.JSON);
    
  } catch (error) {
    return ContentService.createTextOutput(
      JSON.stringify({status: 'error', message: error.toString()})
    ).setMimeType(ContentService.MimeType.JSON);
    
  } finally {
    lock.releaseLock();
  }
}

function checkIfAlreadySubmitted(ss, nama, segmen) {
  var sheet = ss.getSheetByName(segmen);
  if (!sheet) return false;
  
  var data = sheet.getDataRange().getValues();
  for (var i = 1; i < data.length; i++) {
    var rowNama = data[i][1];
    var rowStatus = data[i][3];
    if (rowNama === nama && rowStatus === 'COMPLETED') {
      return true;
    }
  }
  return false;
}
